535 bool contains(const BasicJsonType* ptr) const
536 {
537 for (const auto& reference_token : reference_tokens)
538 {
539 switch (ptr->type())
540 {
541 case detail::value_t::object:
542 {
543 if (!ptr->contains(reference_token))
544 {
545 // we did not find the key in the object
546 return false;
547 }
548
549 ptr = &ptr->operator[](reference_token);
550 break;
551 }
552
553 case detail::value_t::array:
554 {
555 if (JSON_HEDLEY_UNLIKELY(reference_token == "-"))
556 {
557 // "-" always fails the range check
558 return false;
559 }
560 if (JSON_HEDLEY_UNLIKELY(reference_token.size() == 1 && !("0" <= reference_token && reference_token <= "9")))
561 {
562 // invalid char
563 return false;
564 }
565 if (JSON_HEDLEY_UNLIKELY(reference_token.size() > 1))
566 {
567 if (JSON_HEDLEY_UNLIKELY(!('1' <= reference_token[0] && reference_token[0] <= '9')))
568 {
569 // first char should be between '1' and '9'
570 return false;
571 }
572 for (std::size_t i = 1; i < reference_token.size(); i++)
573 {
574 if (JSON_HEDLEY_UNLIKELY(!('0' <= reference_token[i] && reference_token[i] <= '9')))
575 {
576 // other char should be between '0' and '9'
577 return false;
578 }
579 }
580 }
581
582 const auto idx = array_index(reference_token);
583 if (idx >= ptr->size())
584 {
585 // index out of range
586 return false;
587 }
588
589 ptr = &ptr->operator[](idx);
590 break;
591 }
592
593 case detail::value_t::null:
594 case detail::value_t::string:
595 case detail::value_t::boolean:
596 case detail::value_t::number_integer:
597 case detail::value_t::number_unsigned:
598 case detail::value_t::number_float:
599 case detail::value_t::binary:
600 case detail::value_t::discarded:
601 default:
602 {
603 // we do not expect primitive values if there is still a
604 // reference token to process
605 return false;
606 }
607 }
608 }
609
610 // no reference token left means we found a primitive value
611 return true;
612 }

623 static std::vector<std::string> split(const std::string& reference_string)
624 {
625 std::vector<std::string> result;
626
627 // special case: empty reference string -> no reference tokens
628 if (reference_string.empty())
629 {
630 return result;
631 }
632
633 // check if nonempty reference string begins with slash
634 if (JSON_HEDLEY_UNLIKELY(reference_string[0] != '/'))
635 {
636 JSON_THROW(detail::parse_error::create(107, 1, "JSON pointer must be empty or begin with '/' - was: '" + reference_string + "'", BasicJsonType()));
637 }
638
639 // extract the reference tokens:
640 // - slash: position of the last read slash (or end of string)
641 // - start: position after the previous slash
642 for (
643 // search for the first slash after the first character
644 std::size_t slash = reference_string.find_first_of('/', 1),
645 // set the beginning of the first reference token
646 start = 1;
647 // we can stop if start == 0 (if slash == std::string::npos)
648 start != 0;
649 // set the beginning of the next reference token
650 // (will eventually be 0 if slash == std::string::npos)
651 start = (slash == std::string::npos) ? 0 : slash + 1,
652 // find next slash
653 slash = reference_string.find_first_of('/', start))
654 {
655 // use the text between the beginning of the reference token
656 // (start) and the last slash (slash).
657 auto reference_token = reference_string.substr(start, slash - start);
658
659 // check reference tokens are properly escaped
660 for (std::size_t pos = reference_token.find_first_of('~');
661 pos != std::string::npos;
662 pos = reference_token.find_first_of('~', pos + 1))
663 {
664 JSON_ASSERT(reference_token[pos] == '~');
665
666 // ~ must be followed by 0 or 1
667 if (JSON_HEDLEY_UNLIKELY(pos == reference_token.size() - 1 ||
668 (reference_token[pos + 1] != '0' &&
669 reference_token[pos + 1] != '1')))
670 {
671 JSON_THROW(detail::parse_error::create(108, 0, "escape character '~' must be followed with '0' or '1'", BasicJsonType()));
672 }
673 }
674
675 // finally, store the reference token
676 detail::unescape(reference_token);
677 result.push_back(reference_token);
678 }
679
680 return result;
681 }

691 static void flatten(const std::string& reference_string,
692 const BasicJsonType& value,
693 BasicJsonType& result)
694 {
695 switch (value.type())
696 {
697 case detail::value_t::array:
698 {

740 case detail::value_t::binary:
741 case detail::value_t::discarded:

761 static BasicJsonType
762 unflatten(const BasicJsonType& value)
763 {
764 if (JSON_HEDLEY_UNLIKELY(!value.is_object()))
765 {
766 JSON_THROW(detail::type_error::create(314, "only objects can be unflattened", value));
767 }
768
769 BasicJsonType result;
770
771 // iterate the JSON object values
772 for (const auto& element : *value.m_value.object)
773 {
774 if (JSON_HEDLEY_UNLIKELY(!element.second.is_primitive()))
775 {
776 JSON_THROW(detail::type_error::create(315, "values in object must be primitive", element.second));
777 }
778
779 // assign value to reference pointed to by JSON pointer; Note that if
780 // the JSON pointer is "" (i.e., points to the whole value), function
781 // get_and_create returns a reference to result itself. An assignment
782 // will then create a primitive value.
783 json_pointer(element.first).get_and_create(result) = element.second;
784 }
